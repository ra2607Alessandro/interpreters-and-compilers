const jug = 3

for(let i = 0, i <= jug, i = i + 1){
    if (i == jug){
        print(exponential(i))
    }
    print(i)
}

fn exponential(number){
    let times = number
    let result = number
    while (times => 0){
        result = result * number 
        times = times - 1
        if (times == 0){
            print(result)
        }
    }
    
}
